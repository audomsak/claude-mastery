## 3. Database Models

```python
# db/models.py
from sqlalchemy import Column, String, DateTime, Integer, ForeignKey
from sqlalchemy.orm import declarative_base, relationship

Base = declarative_base()

class Tenant(Base):
    __tablename__ = "tenants"
    id = Column(String, primary_key=True)
    name = Column(String)

class User(Base):
    __tablename__ = "users"
    id = Column(String, primary_key=True)
    tenant_id = Column(String, ForeignKey("tenants.id"))
    email = Column(String)
    role = Column(String)  # agent / user / admin

class Ticket(Base):
    __tablename__ = "tickets"
    id = Column(String, primary_key=True)
    tenant_id = Column(String, ForeignKey("tenants.id"), index=True)
    created_by = Column(String, ForeignKey("users.id"))
    assignee = Column(String, ForeignKey("users.id"))
    title = Column(String)
    description = Column(String)
    status = Column(String)  # open / in_progress / resolved / closed
    priority = Column(String)
    created_at = Column(DateTime)
    updated_at = Column(DateTime)
```

⚠️ Every row carries `tenant_id` — used in every query

## 5. Ticket Tools

```python
# tools/tickets.py
from mcp.server import Server
from mcp.types import Tool, TextContent
from mcp.server.auth.middleware import require_auth
from db.connection import get_session
from db.models import Ticket
import uuid

def register(server: Server):
    
    @server.list_tools()
    async def list_tools():
        return [
            Tool(
                name="list_tickets",
                description="List tickets in your tenant. Filters: assignee, status, priority.",
                inputSchema={
                    "type": "object",
                    "properties": {
                        "assignee": {"type": "string", "description": "User ID (default: any)"},
                        "status": {"type": "string", "enum": ["open", "in_progress", "resolved", "closed"]},
                        "priority": {"type": "string", "enum": ["low", "medium", "high", "critical"]},
                        "limit": {"type": "integer", "default": 20, "maximum": 100}
                    }
                }
            ),
            Tool(
                name="create_ticket",
                description="Create a new ticket.",
                inputSchema={
                    "type": "object",
                    "properties": {
                        "title": {"type": "string"},
                        "description": {"type": "string"},
                        "priority": {"type": "string", "enum": ["low", "medium", "high", "critical"]},
                        "assignee": {"type": "string"}
                    },
                    "required": ["title", "description"]
                }
            ),
            # ... more
        ]
    
    @server.call_tool()
    @require_auth(scopes=["tickets.read"])
    async def call_tool(name, args, user_info):
        if name == "list_tickets":
            return await _list_tickets(args, user_info)
        elif name == "create_ticket":
            return await _create_ticket(args, user_info)
        # ...

async def _list_tickets(args, user_info):
    async with get_session() as session:
        # Multi-tenant filter — always
        query = session.query(Ticket).filter(Ticket.tenant_id == user_info.tenant_id)
        
        if args.get("assignee"):
            query = query.filter(Ticket.assignee == args["assignee"])
        if args.get("status"):
            query = query.filter(Ticket.status == args["status"])
        if args.get("priority"):
            query = query.filter(Ticket.priority == args["priority"])
        
        query = query.limit(args.get("limit", 20))
        tickets = await query.all()
        
        # Log access
        audit_log({
            "user": user_info.user_id,
            "tool": "list_tickets",
            "tenant": user_info.tenant_id,
            "filters": args,
            "result_count": len(tickets)
        })
        
        return [TextContent(
            type="text",
            text=json.dumps([{
                "id": t.id,
                "title": t.title,
                "status": t.status,
                "priority": t.priority,
                "assignee": t.assignee
            } for t in tickets])
        )]

async def _create_ticket(args, user_info):
    if "tickets.write" not in user_info.scopes:
        raise PermissionDenied("tickets.write required")
    
    async with get_session() as session:
        new_ticket = Ticket(
            id=str(uuid.uuid4()),
            tenant_id=user_info.tenant_id,
            created_by=user_info.user_id,
            assignee=args.get("assignee", user_info.user_id),
            title=args["title"],
            description=args["description"],
            priority=args.get("priority", "medium"),
            status="open"
        )
        session.add(new_ticket)
        await session.commit()
        
        audit_log({"action": "create_ticket", "ticket_id": new_ticket.id, "user": user_info.user_id})
        
        return [TextContent(
            type="text",
            text=json.dumps({"id": new_ticket.id, "status": "created"})
        )]
```

## 6. Auth Provider

```python
# auth/oauth_provider.py
from mcp.server.auth.provider import OAuthAuthorizationServerProvider
import httpx
import jwt

class MyOAuthProvider(OAuthAuthorizationServerProvider):
    def __init__(self, idp_url, client_id, client_secret):
        self.idp_url = idp_url
        self.client_id = client_id
        self.client_secret = client_secret
        self.http = httpx.AsyncClient()
    
    async def authorize(self, redirect_uri, state, code_challenge, scope):
        # Redirect to IdP
        return f"{self.idp_url}/authorize?client_id={self.client_id}&redirect_uri={redirect_uri}&code_challenge={code_challenge}&code_challenge_method=S256&state={state}&scope={scope}&response_type=code"
    
    async def exchange_code(self, code, redirect_uri, code_verifier):
        r = await self.http.post(f"{self.idp_url}/token", data={
            "grant_type": "authorization_code",
            "code": code,
            "redirect_uri": redirect_uri,
            "code_verifier": code_verifier,
            "client_id": self.client_id,
            "client_secret": self.client_secret
        })
        return r.json()
    
    async def verify_token(self, token):
        # Verify via IdP introspection
        r = await self.http.post(
            f"{self.idp_url}/introspect",
            data={"token": token},
            auth=(self.client_id, self.client_secret)
        )
        result = r.json()
        
        if not result.get("active"):
            raise InvalidToken
        
        return UserInfo(
            user_id=result["sub"],
            tenant_id=result.get("tenant_id"),
            email=result.get("email"),
            scopes=result.get("scope", "").split(),
            session_id=result.get("sid")
        )
```

## 7. Multi-Tenant Test

```python
# tests/test_isolation.py
async def test_cross_tenant_leak():
    """User from tenant A cannot see tenant B's tickets"""
    # Set up
    await create_ticket(tenant_id="A", title="A's ticket")
    await create_ticket(tenant_id="B", title="B's ticket")
    
    # Login as user from A
    user_a_token = await login_as("user_a", tenant="A")
    
    # Call list_tickets
    result = await call_tool_with_auth("list_tickets", {}, user_a_token)
    tickets = json.loads(result[0].text)
    
    # Verify only A's tickets
    assert all(t["tenant_id"] == "A" for t in tickets) if "tenant_id" in tickets[0] else True
    assert "A's ticket" in [t["title"] for t in tickets]
    assert "B's ticket" not in [t["title"] for t in tickets]
```
